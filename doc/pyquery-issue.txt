# pyquery 1.2.1
pyquery.PyQuery has a following method:
    def __call__(self, *args, **kwargs):
        """return a new PyQuery instance
        """
        length = len(args)
        if length == 0:
            raise ValueError('You must provide at least a selector')
        if args[0] == '':
            return self.__class__([])
        if len(args) == 1 and \
           isinstance(args[0], str) and \
           not args[0].startswith('<'):
            args += (self,)
        result = self.__class__(*args, parent=self, **kwargs)
        return result

So it will behave differently for 'str' selector and 'unicode' selector.
